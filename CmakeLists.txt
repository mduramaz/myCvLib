cmake_minimum_required(VERSION 3.5)


project(healloascmake)

cuda_add_library(CUDASampLib src/cudatest.cu  OPTIONS ${GENCODE} )

                        src/image.cpp)
target_link_libraries(hello_cmake CUDASampLib)
